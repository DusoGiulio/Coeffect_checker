	Privacy sup(Privacy x) {
		Privacy result;
		if (this.leq(x))
			result = x;
		else
			result = this;
		return result;
	}

		return true;
	}

	Privacy sum(Privacy x) {
		return this.sup(x);
	}

		return this;
	}

		return new Irrelevant();
